# 表单校验/常用规则
---

```js
const rules = {
    username: [
      { required: true, message: '请输入用户名', trigger: 'blur' },
      {
        pattern: /^[A-Za-z0-9]{6,16}$/,
        message: '用户名仅支持6-16位数字/字母组合',
        trigger: 'blur',
      },
    ],
    password: [
      { required: true, message: '请输入密码', trigger: 'blur' },
      {
        pattern: /^([^\u4e00-\u9fa5\s]){6,16}$/,
        message: '密码仅支持6-16位数字/字母/符号组合',
        trigger: 'blur',
      },
    ],
    realName: [
      { required: true, message: '请输入真实名称', trigger: 'blur' },
      {
        pattern: /^([\u4e00-\u9fa5]{2,16}|[A-Za-z\s]{2,16})$/,
        message: '真实名称仅支持2-16位中文或英文名',
        trigger: 'blur',
      },
    ],
    mobile: [
      { required: true, message: '请输入电话号码', trigger: 'blur' },
      {
        pattern: /^((0\d{2,3}-\d{7,8})|(1[3456789]\d{9}))$/,
        message: '号码格式错误',
        trigger: 'blur',
      },
    ],
  };
```
